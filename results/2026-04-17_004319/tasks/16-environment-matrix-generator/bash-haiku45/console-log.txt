generate_cartesian_product() {
    local os_array="$1"
    local version_array="$2"
    local language_array="$3"
    local exclude_rules="$4"

    local matrix='{"include": []}'

    # Get lengths
    local os_len version_len language_len
    os_len=$(jq 'length' <<< "$os_array")
    version_len=$(jq 'length' <<< "$version_array")
    language_len=$(jq 'length' <<< "$language_array")

    echo "Lengths: os=$os_len, version=$version_len, lang=$language_len" >&2

    # If all arrays are empty, return empty matrix
    if [[ $os_len -eq 0 ]] && [[ $version_len -eq 0 ]] && [[ $language_len -eq 0 ]]; then
        echo "$matrix"
        return 0
    fi

    # Iterate through all combinations
    for ((i = 0; i < os_len; i++)); do
        local os_val
        os_val=$(jq -r ".[$i]" <<< "$os_array")
        echo "Processing OS: $os_val" >&2

        for ((j = 0; j < version_len; j++)); do
            local version_val
            version_val=$(jq -r ".[$j]" <<< "$version_array")
            echo "  Processing version: $version_val" >&2

            for ((k = 0; k < language_len; k++)); do
                local language_val
                language_val=$(jq -r ".[$k]" <<< "$language_array")
